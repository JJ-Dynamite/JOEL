## Types

### Result Type

```joel
# Coming soon
fn divide(a: f64, b: f64) -> Result<f64, str> {
  if b == 0.0 {
    return Err("Division by zero")
  }
  return Ok(a / b)
}

match divide(10.0, 2.0) {
  Ok(result) => print("Result:", result),
  Err(error) => print("Error:", error)
}
```

### Option Type

```joel

fn find_item(items: list[str], target: str) -> Option[str> {
  for item in items {
    if item == target {
      return Some(item)
    }
  }
  return None
}
```
